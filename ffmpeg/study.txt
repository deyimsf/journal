#命令参数基本可以分成5部分
$ffmpeg [全局参数] [输入文件参数] -i [输入文件] [输出文件参数] [输出文件]

#可以写成多行
$ffmpeg  \
		[全局参数]  \
		[输入文件参数]  \
		-i  [输入文件]  \
		[输出文件参数]  \
		[输出文件]

#截取视频,从26秒开始，最多截取8秒，拷贝视频编码，拷贝音频编码，输出到output.mp4
$ffmpeg -ss 26 -t 8   -i  1234.mp4   -c:v copy -c:a copy     output.mp4
 或
$ffmpeg \
      -y \ 				     //全局参数，不经过确认，输出时直接覆盖同名文件
	  -ss 26 -t 8 \          //输入文件参数
	  -i 1234.mp4 \          //输入文件
	  -c:v copy -c:a copy \  //输出文件参数
	  output.mp4             //输出文件
